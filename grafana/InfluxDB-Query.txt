from(bucket: "BucketMeaserments")
  |> range(start: v.timeRangeStart, stop: v.timeRangeStop)
  |> filter(fn: (r) => r._measurement == "temperature")
  |> filter(fn: (r) => r._field == "value")
  |> filter(fn: (r) => r.alias == "esp32-XXXXXX")
  |> group(columns: ["sensor","room"])
  |> sort(columns: ["_time"])
  |> yield()
